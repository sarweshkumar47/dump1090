# -----------------------------------------------------------------------------
# DUMP1090
# ========
#
# This dump1090_parameters.txt file contains a number of parameters that are
# required to ensure the DUMP1090 demo works as requried.
#
# The list of parameters below, indicated by the prefix DUMP1090 at the start
# of each line are those which will be picked up by the various components of
# the DUMP1090 demo.
#
# The structure of a paramater line is as follows:
#
#    Parameter Name (e.g. DUMP1090_PARA)
#    Space or Tab
#    Parameter Value (e.g. This the para value)
#
# NOTE: Do not enclose the Parameter Value in quotation marks (unless a
# quotation mark is required in the value).  All characters following the
# initial space/tab to the end of the line will be picked up as the parameter
# value.
#
# All other lines (blank lines, lines beginning with the hash (#) symbol
# (comments) and lines NOT beginning with DUMP1090 will be ignored.
#
# Each DUMP1090 parameter below will have one or more comment lines providing
# a description of the parameter.
# 
# To help navigate this parameter file, the paramaters have been grouped as
# follows:
#
#    1. DUMP1090 demo general Settings (username, temporary directories etc)
#
#    2. RASPBERRY PI settings (from where the aircraft ADSB streaming data is
#    captured and decoded.
#
#    3. KAFKA settings (details about where the KAFKA messaging system can be
#    found)
#
#    4. DATABASE settings (identifying the Ingres/Vector instance and database
#    in which the DUMP1090 data is both stored and retrieved).
#
#    5. VISUALIZATION settings 
#
# -----------------------------------------------------------------------------

# =============================================================================
# -----------------------------------------------------------------------------
# DUMP1090 General Settings
# -----------------------------------------------------------------------------
# =============================================================================
#
# -----------------------------------------------------------------------------
# DUMP1090_BASE_DIR: The top level directory under which the DUMP1090 softare
# and files have been placed.
# -----------------------------------------------------------------------------
DUMP1090_BASE_DIR /home/actian/projects/dump1090/dump1090_demo

# -----------------------------------------------------------------------------
# DUMP1090_TMP_DIR: The dump1090 application will create a number of
# relatively small temporary and work files (which SHOULD automatically be
# removed as part of the normal operation of the demo).
# This parameter should point to a directory in which these temporary files
# can be written.  The directory should have sufficient space (~10MB) and have
# permissions such that the user (see DUMP1090_USER) can read/write.
# -----------------------------------------------------------------------------
DUMP1090_TMP_DIR /tmp

# -----------------------------------------------------------------------------
# DUMP1090_LOG: A Y/N flag to indicate whether the DUMP1090 should retain
# its logging information.  This is useful when trying to determine why
# DUMP1090 is not working correctly.
# 
# When logging is enabled (Y), various log files will be placed in the "log"
# directory under the DUMP1090_BASE_DIR directory.
# -----------------------------------------------------------------------------
DUMP1090_LOG Y

# -----------------------------------------------------------------------------
# DUMP1090_VERBOSE: A Y/N flag to indicate whether the DUMP1090 should report
# messages that are written to LOG file (via MESSAGELOG) are also to be
# displayed on standard output.
# -----------------------------------------------------------------------------
DUMP1090_VERBOSE Y

# -----------------------------------------------------------------------------
# DUMP1090_USER: This is the operating system user that will be running the
# demo.  Typically this will be the owner of the database in which the
# DUMP1090 data is stored, and from which the visualization components are
# run.
# -----------------------------------------------------------------------------
DUMP1090_USER actian


# =============================================================================
# -----------------------------------------------------------------------------
# RASPBERRY PI Settings
# -----------------------------------------------------------------------------
# =============================================================================

# -----------------------------------------------------------------------------
# DUMP1090_RPI_IP: The DNS or IP address of the Raspberry Pi that is capturing
# and decoding the ADSB aircraft transmissions.
# -----------------------------------------------------------------------------
DUMP1090_RPI_IP pennardpi01.etowns.org

# -----------------------------------------------------------------------------
# DUMP1090_RPI_PORT: The port number through which the Raspberry Pi is feeding
# the decoded ADSB aircraft transmissions.
# -----------------------------------------------------------------------------
DUMP1090_RPI_PORT 30003


# =============================================================================
# -----------------------------------------------------------------------------
# KAFKA Settings
# -----------------------------------------------------------------------------
# =============================================================================

# -----------------------------------------------------------------------------
# DUMP1090_KAFKA_BROKER_IP: The DNS or IP address of the KAFKA broker service.
# -----------------------------------------------------------------------------
DUMP1090_KAFKA_BROKER_IP vortex01

# -----------------------------------------------------------------------------
# DUMP1090_KAFKA_BROKER_PORT: The port number of the KAFKA broker service.
# -----------------------------------------------------------------------------
DUMP1090_KAFKA_BROKER_PORT 6667

# -----------------------------------------------------------------------------
# DUMP1090_KAFKA_TOPIC: The kafka topic which will be used by DUMP1090
# -----------------------------------------------------------------------------
DUMP1090_KAFKA_TOPIC adsb_dump1090



# =============================================================================
# -----------------------------------------------------------------------------
# DATABASE Settings
# -----------------------------------------------------------------------------
# =============================================================================

# -----------------------------------------------------------------------------
# DUMP1090_II_SYSTEM: The value of II_SYSTEM for the Ingres/Vector/Vector-H
# instance that houses the DUMP1090 database
# -----------------------------------------------------------------------------
DUMP1090_II_SYSTEM /opt/Actian/IngresT2

# -----------------------------------------------------------------------------
# DUMP1090_IVH: An indication to what type of instance this demo is 
#   ingres
#   vector
#   vector-h
# -----------------------------------------------------------------------------
DUMP1090_IVH ingres

# -----------------------------------------------------------------------------
#DUMP1090_DATABASE: The name of the database in which dump1090 data will be
# stored.
# -----------------------------------------------------------------------------
DUMP1090_DATABASE dump1090

# -----------------------------------------------------------------------------
DUMP1090_SQL_TIME_SLICE_UNITS minutes
